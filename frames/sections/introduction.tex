\begin{frame}{Problem Definition}{}
    \HL
    \only<-3>{(\HLshort)}%
    \only<4->{(\alert{\HLdeg})}

    \uncover<+->{
        \uncover<+->{
            \emph{Instance}: $(G, L, b, c, d)$, where
            \begin{itemize}
                \item $G = (V,E)$ is a network
                \item $L \subseteq V$ are leaders, $F = V \smallsetminus L$ are followers
                \item $b$ is a budget
                \item \only<-3>{$c : G \times V \rightarrow \mathbb{R}$ is a centrality measure}
                \only<4->{\alert{$c = c_{deg}$, $c_{deg}(G, v) = \deg(v)$}}
                \item $d$ is a safety margin
            \end{itemize}
        }

        \uncover<+->{
            \emph{Question}: Is there $W \subseteq F \times F$ and $F' \subseteq F$ such that
            \begin{itemize}
                \item $|F'| \geq d$
                \item $|W| \leq b$
                \item \only<-3>{$\forall_{f' \in F'} \forall_{\ell \in L} \colon c\big((V, E \cup W), f'\big) \geq c\big((V, E \cup W), \ell\big)$}
                \only<4->{\alert{$\forall_{f' \in F'} \colon \deg(f') \geq \max\limits_{\forall_{\ell \in L}}\deg(\ell)$ in $(V, E \cup W)$}}
            \end{itemize}
        }
    }
\end{frame}
